CREATE TABLE study (
    id INT,
    server TEXT,
    alt_server TEXT,
    runs INT,
    total_samples INT,
    samples_per_page INT,
    repetitions INT,
    shuffle BOOL,
    timestamp DATETIME
);

CREATE TABLE samples (
    study_id INT,
    subject TEXT,
    predicate TEXT,
    object TEXT
);

CREATE TABLE results (
    study_id INT,
    elapsed TIME(6),
    pattern TEXT,
    perPage INT,
    server TEXT,
    total_items INT,
    vars INT,
    timestamp DATETIME
);
